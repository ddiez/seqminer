        def initialize(name, type, options = {:config => nil})
            super()

            if ! options[:config]
                @config = SeqMiner::Config.new
            else
                @config = options[:config]
            end
            @dir = config.dir_sequence + name
            @type = type
            @file = @dir + (type + ".fa")
            fi = Bio::FastaFormat.open(@file)
            fi.each do |entry|
                seq = entry.to_biosequence
                case type
                when "gene"
                    seq.na
                when "protein"
                    seq.aa
                end
                add(seq, seq.accessions[0])
            end
        end
