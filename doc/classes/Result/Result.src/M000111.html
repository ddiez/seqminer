                def write_fasta_spp
                        genome = Genome::Set.new(taxon, options = {:config => config})
                        genome.filter_by_acc(items.keys)
                        ["protein", "cds", "gene"].each do |type|
                                file = config.dir_result + "genome/fasta" + ortholog.name + (id + "_" + type + ".fa")
                                genome.write_fasta(type, file)
                        end
                end
