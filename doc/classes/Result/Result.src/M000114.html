                def debug
                        warn "+ Result +"
                        warn "* result id: " + id
                        warn "* tool: " + tool.to_s
                        warn "* sequences: " + length.to_s
                        nsh = 0
                        ndom = 0
                        ndomh = 0
                        each_sequence do |hit|
                                nsh += hit.length
                                hit.each_sequencehit do |subhit|
                                        ndom += subhit.length
                                        subhit.each_domain do |domain|
                                                ndomh += domain.length
                                        end
                                end
                        end
                        warn "* sequence hits: " + nsh.to_s
                        warn "* domains: " + ndom.to_s
                        warn "* domain hits: " + ndomh.to_s
                end
