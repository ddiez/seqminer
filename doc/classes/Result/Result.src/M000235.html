                def filter_domain_by_eval(val = 0.001)
                        each_sequence do |seq|
                                seq.each_sequencehit do |sh|
                                        dd = []
                                        sh.each_domain do |domain|
                                                dd << domain if domain.eval > val
                                        end
                                        dd.each do |i|
                                                sh.delete(i)
                                        end
                                        seq.delete(sh) if sh.length == 0
                                end
                                delete(seq) if seq.length == 0
                        end
                end
