# File Result.rb, line 39
        def localization
            loc = []
            # we have to sort the hash in this case.
            @items.sort.each_index do |index|
                loc << @items[index+1].aln_from.to_s + ".." + @items[index+1].aln_to.to_s
            end
            loc.join(",")
        end

        def max_completeness
            mc = 0

                smc = domain.completeness
                mc = smc if smc > mc
            end
            mc
        end
