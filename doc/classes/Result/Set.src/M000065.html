                def best_hit(limit = nil)
                        bh = nil
                        items.each_value do |result|
                                if ! limit.nil?
                                        next if ! limit.include?(result.type)
                                end
                                ch = result.best_hit
                                bh = ch if bh.nil? or ch.eval < bh.eval
                        end
                        return bh
                end
