        def test_add
                is = Item::Set.new
                i1 = Item::Item.new(0)
                i2 = Item::Item.new(2)
                assert_not_nil(is.add(i1))
                assert_not_nil(is << i2)
                assert_equal(2, is.length)
                assert_raise RuntimeError do
                        is << i2
                end
        end
