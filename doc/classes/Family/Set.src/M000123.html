                def initialize(options = {:config => config})
                        super()
                        
                        if options[:config]
                                @config = options[:config]
                        else
                                @config = Config::General.new
                        end
                        
                        file = config.dir_config + "family.txt"
                        fo = File.open(file)
                        fo.each_line do |line|
                                next if line.match(/^#/)
                                line.chomp!
                                t, o, fn = line.split("\t")
                                family = Family.new(t + "-" + o)
                                family.name = fn
                                family.taxon = t
                                family.ortholog = o
                                self << family
                        end
                        fo.close
                end
