                def initialize(ps, options = {:empty => false, :config => nil})
                        super()
                        
                        if ! options[:config]
                                @config = Config::General.new
                        else
                                @config = options[:config]
                        end

                        if ! options[:empty]
                                if ! ps.nil?
                                        @parameters = ps
                                        populate
                                else
                                        raise "ERROR: need a valid ParameterSet object"
                                        exit
                                end
                        end
                end
