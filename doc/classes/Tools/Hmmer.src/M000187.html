                def build_cmd
                        case tool
                        when 'hmmsearch'
                                cmd = [path, parameters, "--domtblout", table_file, model, infile, ">", outfile]
                        when 'hmmscan'
                                cmd = [path, parameters, "--domtblout", table_file, model, infile, ">", outfile]
                        when 'hmmalign'
                                cmd = [path, parameters, model, infile, ">", outfile]
                        when 'hmmpress'
                                cmd = [path, parameters, model, infile, ">", outfile]
                        end
                        @cmd = cmd.join(" ")
                end
