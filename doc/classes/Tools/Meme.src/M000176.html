                def build_cmd
                        case tool
                        when 'meme'
                                cmd = [path, infile, "-o", outdir, parameters]
                        when 'mast'
                                cmd = [path, infile, "-d", db, parameters] 
                        when 'tomtom'
                                cmd = [path, parameters, "-query", query, "-target", target]
                        end
                        @cmd = cmd.join(" ")
                end
