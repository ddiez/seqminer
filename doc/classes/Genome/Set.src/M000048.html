                def write_table(file = nil)
                        if file
                                fo = File.new(file, "w")
                        else
                                fo = $stdout
                        end
                        
                        fo.puts "id\t" +
                                "chromosome\t" +
                                "source\t" +
                                "type\t" +
                                "exon\t" +
                                "strand\t" +
                                "from\t" +
                                "to\t" +
                                "pseudogene\t" +
                                "references\t" +
                                "mol_type\t" +
                                "description"
                        each_gene do |gene|
                                fo.puts gene.id + "\t" +
                                        gene.chromosome + "\t" +
                                        gene.source + "\t" +
                                        "gene" + "\t" +
                                        "\t" +
                                        gene.strand.to_s + "\t" +
                                        gene.from.to_s + "\t" +
                                        gene.to.to_s + "\t" +
                                        gene.pseudogene.to_s + "\t" +
                                        gene.references.to_s + "\t" +
                                        gene.type + "\t" +
                                        gene.description
                                gene.each_exon do |exon|
                                        fo.puts gene.id + "\t" +
                                                "\t" +
                                                "\t" +
                                                "exon" + "\t" +
                                                exon.id.to_s + "\t" +
                                                exon.strand.to_s + "\t" +
                                                exon.from.to_s + "\t" +
                                                exon.to.to_s + "\t" +
                                                "\t" +
                                                "\t" +
                                                "\t" +
                                                ""
                                end
                        end
                        fo.close
                end
