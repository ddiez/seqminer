                def splicing_original
                        loc = []
                        case strand
                        when 1
                                exons = ids
                        when -1
                                exons = ids.reverse
                        end
                        
                        exons.each do |exon|
                                x0 = items[exon].from
                                x1 = items[exon].to
                                loc << x0.to_s + ".." + x1.to_s
                        end
                        loc = loc.join(",")
                        loc = "join(" + loc + ")" if length > 1
                        loc = "complement(" + loc + ")" if strand == -1
                        loc
                end
