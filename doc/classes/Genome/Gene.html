        def splicing
            loc =[]
            
            case strand
            when 1
                exons = oitems
            when -1
                exons = oitems.reverse
            end
            
            f0 = exons[0].from
            exons.each do |exon|
                x0 = exon.from - f0 + 1
                x1 = exon.to - f0 + 1
                loc << x0.to_s + ".." + x1.to_s
            end
            loc = loc.join(",")
            loc = "join(" + loc + ")" if length > 1
            loc = "complement(" + loc + ")" if strand == -1 # in the current version this is not needed.
            loc
        end
