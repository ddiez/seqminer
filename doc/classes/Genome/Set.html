def _parse_desc(line) 

    valid_fields = ["description", "pseudogene"]
    a = line.split(/=|;/)
    h = {}
    valid_fields.each do |field|
        fi = a.index(field)
        if ! fi.nil?
            h[field] = a[fi + 1]
        end
    end
    h

end
