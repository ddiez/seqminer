update_pssm()

This method computes the best hit for a series of genomic searches belonging to the same species and then returns the best hit (for protein searched) as seed for a PSI-Blast search.

[Source]

     # File seqminer.rb, line 218
218:         def update_pssm
219:             typeset = Search::TypeSet.new
220:             ortholog.each_ortholog do |o|
221:                 o.debug
222:                 rs = Result::Set.new
223:                 warn "* computing best hit"
224:                 taxon.each_taxon do |t|
225:                     next if t.type != 'spp'
226:                     typeset.each_value do |type|
227:                         next if type.name != "protein"
228:                         sid = t.name + "-" + o.name + "-" + type.name
229:                         file = config.dir_result + "genome/search" + o.name + (sid + ".txt") 
230:                         p = Result::HmmerParser.new(options = {:config => config, :taxon => t, :ortholog => o})
231:                         p.file = file
232:                         p.result_id = sid
233:                         p.type = type
234:                         r = p.parse
235:                         rs << r
236:                     end
237:                 end
238:                 #rs.debug
239:                 rs.filter_by_eval(0.001)
240:                 rs.clean_up
241:                 next if rs.length == 0
242:                 #o.debug
243:                 #rs.debug
244:                 bh = rs.best_hit
245:                 bh.debug
246:                 bh.taxon.debug
247:

253:                 dir.mkpath if ! dir.exist?

256:                 fo.puts seq.seq.to_fasta(seq.definition, 60)
257:                 fo.close
258:                 
259:                 # Run the search with the SEED and the genome.
260:                 warn "* computing PSSM model for Blast+"
261:                 pssm_file = dir + (bh.ortholog.name + ".pssm")
262:                 pgp_file = dir + (bh.ortholog.name + ".pgp")
263:                 ts = Tools::Blast.new('psiblast', options = {:config => config})
264:                 ts.db = config.dir_sequence + bh.taxon.name + "protein"
265:                 ts.seed_file = file
