                def filter_by_name(filter)
                        dor = []
                        each_ortholog do |ortholog|
                                m = false
                                filter.each do |f|
                                        if ortholog.name.match(/#{f}/)
                                                m = true
                                                break
                                        end
                                end
                                dor << ortholog if ! m
                        end
                        dor.each do |i|
                                delete(i)
                        end
                end
