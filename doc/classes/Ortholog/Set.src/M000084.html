                def initialize(options = {:empty => false, :config => nil})
                        super()

                        if ! options[:config]
                                @config = Config::General.new
                        else
                                @config = options[:config]
                        end
                        
                        if ! options[:empty]
                                of = config.file_ortholog
                                file = File.open(of)
                                file.each do |line|
                                        line = line.chomp
                                        next if line == ""
                                        next if line =~ /^#/
                                        name, hmm = line.split("\t")
                                        #warn "* Ortholog: " + name
        
                                        ortholog = Ortholog.new(name, hmm)
                                        self.add(ortholog)
                                end
                                file.close
                        end
                end
