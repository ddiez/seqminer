                def add(*args)
                        case args.length
                        when 1
                                item = args[0]
                                id = item.id
                        when 2
                                item = args[0]
                                id = args[1]
                        end
                        
                        if items.has_key?(id)
                                raise "Item with id #{id} already exists, use replace instead!"
                        else 
                                items[id] = item
                        end
                end
