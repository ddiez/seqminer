                def filter_by_type(filter)
                        dt = []
                        each_taxon do |taxon|
                                m = false
                                filter.each do |f|
                                        if taxon.type.match(/#{f}/)
                                                m = true
                                                break
                                        end
                                end
                                dt << taxon if ! m
                        end
                        dt.each do |taxon|
                                delete(taxon)
                        end
                end
