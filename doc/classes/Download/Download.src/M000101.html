                def download_nucest
                        outdir = config.dir_source + taxon.name
                        outdir.mkpath if ! outdir.exist?
                        outfile = outdir + (taxon.name + "_nucest.gb")
                        outfile.unlink if outfile.exist?
                        
                        term = "txid" + taxon.id + "[Organism:exp]"
                        
                        ncbi_download(term, "nucest", outfile)
                end
