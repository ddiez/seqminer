                def download_spp
                        puts "* downloading spp " + id
                        case taxon.source
                        when "plasmodb"
                                download_eupathdb('plasmodb')
                        when "tritrypdb"
                                download_eupathdb('tritrypdb')
                        when "giardiadb"
                                download_eupathdb('giardiadb')
                        when "broad"
                                download_broad
                        when "ncbi"
                                download_ncbi
                        end
                end
