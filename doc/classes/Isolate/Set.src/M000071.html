                def write_table(file = nil)
                        if file
                                fo = File.new(file, "w")
                        else
                                fo = $stdout
                        end
                        
                        fo.puts "id\t" +
                                "accession\t" +
                                "subid\t" +
                                "locus\t" +
                                "division\t" +
                                "source\t" +
                                "isolate\t" +
                                "strain\t" +
                                "clone\t" +
                                "country\t" +
                                "type\t" +
                                "pseudogene\t" +
                                "strand\t" +
                                "from\t" +
                                "to\t" +
                                "trans_table\t" +
                                "references\t" +
                                "description"
                        each_sequence do |seq|
                                fo.puts seq.id + "\t" +
                                        seq.accession + "\t" +
                                        seq.subid + "\t" +
                                        seq.locus + "\t" +
                                        seq.division + "\t" +
                                        seq.source + "\t" +
                                        seq.isolate.to_s + "\t" +
                                        seq.strain.to_s + "\t" +
                                        seq.clone.to_s + "\t" +
                                        seq.country.to_s + "\t" +
                                        seq.type + "\t" +
                                        seq.pseudogene.to_s + "\t" +
                                        seq.strand.to_s + "\t" +
                                        seq.from.to_s + "\t" +
                                        seq.to.to_s + "\t" +
                                        seq.trans_table.to_s + "\t" +
                                        seq.references + "\t" +
                                        seq.description
                        end
                        fo.close
                end
